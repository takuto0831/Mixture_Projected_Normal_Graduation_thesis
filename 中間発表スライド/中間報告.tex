\section{Black--Littermanモデル}
\begin{frame}{Black--Littermanモデル}
Black--Littermanモデルは，市場均衡における期待収益率と投資家の見通しを組み合わせて投資比率を求める手法である．
\begin{enumerate}
\setlength{\itemsep}{0.5cm}
\item
金融市場の均衡における期待収益率である均衡期待収益率を求める．\\
\begin{itembox}[l]{均衡期待収益率$\Pi$は以下の式で求める．}
%均衡期待収益率$\Pi$は以下の式で求める．
\vspace{-1zh}

%\vspace{-1.5zh}
\begin{itemize}
\setlength{\itemsep}{-0.2cm}
\item $\lambda$：リスク回避度 \\
\item $\Sigma$：期待収益率の分散共分散行列  \\
\item $w_{mkt}$：時価総額比率\\
\end{itemize}
$$\Pi = \lambda\Sigma w_{mkt}.$$
\end{itembox}
時価総額比率を用いることで市場均衡を反映させる．
\item
投資家の収益性に関する見通しを均衡期待収益率にブレンドする．\\
\begin{itemize}
\item \textcolor{red}{絶対的見通し}：ある一つの資産の収益の変化を示す見通し

\item 相対的見通し：複数の資産の相関を示す見通し
\end{itemize}
\item
最適化計算により最適投資比率を求める．
\end{enumerate}
\end{frame}

\begin{frame}{均衡期待収益率}
期待収益率の不確実性を考慮するため誤差項$\varepsilon_\mu$を加え，期待収益率$\mu$は以下の式で表される．
{\large $$\mu = \Pi + \varepsilon_\mu,\  \varepsilon_\mu \sim N(0,\tau\Sigma),$$}
%\vspace{1zh}

ここで，$\tau$は市場均衡の信頼度で，定義域は$0\leq\tau\leq 1$である．\\
\begin{itemize}
\item $\tau$が小さい$\rightarrow$市場の均衡が成立している
\item $\tau$が大きい$\rightarrow$市場が均衡状態から乖離している
\end{itemize}
以上より，Black--Littermanモデルの期待収益率$\mu$の分布は次の正規分布に従う．
{\large $$\mu \sim N(\Pi,\tau\Sigma).$$}
\end{frame}

\begin{frame}{投資家の見通し (1/2)}
どの資産に対しての見通しであるかを表す行列$P$，投資家の期待収益率における見通しを表す行列$Q$を用意する．
見通しの数が$k$，資産数が$n$とする．
\vspace{-0.5zh}
{\small \begin{equation*}
P = \begin{pmatrix}
                     p_{1,1} & \dots & p_{1,n} \\
			 \vdots & \ddots &\vdots \\
			  p_{k,1} & \dots & p_{k,n}
            \end{pmatrix},\ 
Q = \begin{pmatrix}
                     v_1 \\
                     v_2 \\
                     \vdots  \\
			  v_k
            \end{pmatrix}.
\end{equation*}}
$P$は$k\times n$行列，$Q$は$k\times 1$行列である．\\
%\vspace{0.5zh}

\begin{itembox}[l]{例}
A，B，Cの3つの資産があり，2つの見通しを加えるとする．
\begin{enumerate}
\item Aの資産が絶対的に5\%の収益を持つ．%（自信度：100\%）
\item Bの資産が絶対的に2\%の収益を持つ．%（自信度：100\%）
\end{enumerate}
このとき，見通しの行列$P$,$Q$は次のように表すことができる．
\vspace{-0.5zh}
\begin{equation*}
P=
\begin{pmatrix}
1 &0 &0 \\
0 &1 &0
\end{pmatrix},
Q=
\begin{pmatrix}
0.05 \\
0.02
\end{pmatrix}.
\end{equation*}
\end{itembox}
\end{frame}



\begin{frame}{投資家の見通し (2/2)}
投資家の見通しの不確実性を考慮するため誤差項$\varepsilon_v$を見通しに加える．すると，見通しの情報は以下の式で表される．
{\large $$P\mu = Q + \varepsilon_v,\ \ \ \varepsilon_v \sim N(0,\Omega),$$}

ここで，$\Omega$は見通しの不確実性を表す分散共分散行列である．\\そして，投資家の見通しの分布は次の正規分布に従う．
{\large $$P\mu \sim N(Q,\Omega).$$}
\end{frame}

\begin{frame}{Black--Littermanモデル}
事前均衡分布と投資家の見通しは次の$n$，$k$変量正規分布に従う．
$$f(x;\mu)=\frac{1}{(\sqrt{2\pi})^n\sqrt{|\tau\Sigma|}}\exp\left(-\frac{1}{2}(x-\Pi)^{T}(\tau\Sigma)^{-1}(x-\Pi)\right),$$
$$f(x;P\mu)=\frac{1}{(\sqrt{2\pi})^k\sqrt{|\Omega|}}\exp\left(-\frac{1}{2}(x-Q)^{T}\Omega^{-1}(x-Q)\right).$$

\vspace{0.5zh}
ベイズの定理より期待収益率の分布を尤度，投資家の見通しの分布を事前分布とし，Black--Littermanモデルの分布である事後分布を求める．\\
Black--Littermanモデルの期待収益率$\mu_{BL}$，分散共分散行列$\Sigma_{BL}$は以下のようになる．
%ベイズの定理よりBlack--Littermanモデルの期待収益率$\mu_{BL}$，分散共分散行列$\Sigma_{BL}$は以下のようになる．
{\Large $$\mu_{BL}= [(\tau \Sigma)^{-1}+P^{T}\Omega^{-1}P]^{-1}[(\tau \Sigma)^{-1} \Pi+P^{T}\Omega^{-1}Q],$$}
\vspace{-1zh}
{\Large $$\Sigma_{BL} = [(\tau \Sigma)^{-1} + P^T\Omega^{-1}P]^{-1}.$$}
\end{frame}




\begin{frame}{最適化理論}
Black--Littermanモデルに用いた最適化計算を以下の制約条件を加えて定式化する．
\begin{equation*}
\begin{aligned}
& \mbox{最小化} & &\frac{1}{2}\lambda w^T \Sigma_{BL} w - \mu_{BL}^Tw, \\
& \mbox{制約条件} & &\sum_{i=1}^{n}w_i = 1, \\
& & &\min(w_{mkt}) \leq w_i \leq \max(w_{mkt}), (i=1,\dots, n).\\
\end{aligned}
\end{equation*}

\begin{itemize}
\item 投資比率の合計を1とする．

\item 投資比率の上限，下限をそれぞれ時価総額比率の最大値と最小値に設定する．
\end{itemize}
\end{frame}

\section{データ解析}
\begin{frame}{実装方法}
\begin{itemize}
\setlength{\itemsep}{0.5cm}
\item 実装手順\\
\begin{enumerate}
\item 実際の株価の収益率から分散共分散行列$\Sigma$を求める．

\item 時価総額比率$w_{mkt}$と$\Sigma$を用いて，均衡期待収益率$\Pi$を求める．

\item 見通しの情報となる行列$P$，$Q$を作成する．

\item 見通しを加味したBlack--Littermanモデルの平均$\mu_{BL}$，分散$\Sigma_{BL}$を求める．

\item 最適化計算により最適投資比率を求める．

\end{enumerate}

\item 投資家の見通し\\
目標株価，予想EPS，予想PERの3つの見通しデータを用いて，Black--Littermanモデルに当てはめた．
%\item リバランス頻度\\
%リバランスする期間を1週間，2週間，1ヶ月，2ヶ月，3ヶ月，半年，1年と設定した．
\item $\tau$と$\lambda$の設定\\
\begin{itemize}
\item $\tau$はBlamont\ and\ Firoozy (2003)の例を採用し，$\tau=1/n$（$n$は観測数）と定義する．

\item $\lambda$はGrinold\ and\ Kahn (1999)の例を採用し，$\lambda = E(r)-r_f/\sigma^2$と定義した．$E(r)$はリスクプレミアム，$r_f$はリスクフリーレート，$\sigma^2$はマーケットリターンの分散である．%計算の結果から$\lambda=1.89$と設定した．
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{モデル}
$\sigma_{ij}$は資産$i$と資産$j$の収益率の共分散，$w_i$を資産$i$のポートフォリオウェイト，$w_j$を資産$j$のポートフォリオウェイト，$R_p$はポートフォリオの期待収益率，$R_E$は投資家の要求期待収益率，$n$は資産数とする．\\
\vspace{0.5zh}
平均分散モデルとは以下のようなモデルである．
\begin{eqnarray*}
{\rm minimize}&&\sum_{i=1}^{n} \sum_{j=1}^{n} \sigma_{ij}w_iw_j,\\
{\rm subject\ to}&&R_p \geq R_E,\\
 &&\sum_{i=1}^{n}w_i = 1.\\
\end{eqnarray*}
\end{frame}


\begin{frame}{平均分散アプローチによるポートフォリオ選択問題}
平均分散アプローチでの，最適ポートフォリオ選択問題は

{\large$$ \mbox{最小化} \ \frac{1}{2}\lambda w^{T}\Sigma w - w^{T}\mu$$}

と表すことができその解は
{\large $$ w^{*} = \frac{1}{\lambda}\Sigma^{-1}\mu $$}
となる．この$w^{*}$が最適投資比率になる．
\end{frame}

\begin{frame}{見通しの不確実性$\Omega$}
各見通しの誤差項を分散$\omega_i(i=1,\dots,K)$とすると，$\Omega$は以下の式で表される．ここで，見通しが互いに独立であると仮定しているため，$\Omega$の非対角成分は全て0である．
$$
\Omega = 
\begin{bmatrix}
\omega_1  &0 &0     \\
0 &\ddots & 0  \\
0 & 0 & \omega_K
\end{bmatrix}
$$
He and Litterman (1999)は$\omega_i/\tau$の比率は見通しのポートフォリオの分散$p_i\Sigma p_i'$と等しいと仮定し，$\Omega$を次のように表した．
$$
\Omega = 
\begin{bmatrix}
\tau \cdot p_1\Sigma p_1'  &0 & \dots &0     \\
0 & \tau \cdot p_2\Sigma p_2' & \ddots & \vdots \\
\vdots & \ddots & \ddots & 0  \\
0 & \dots & 0 &\tau \cdot p_k\Sigma p_k'
\end{bmatrix}
$$
\end{frame}

\begin{frame}{平均分散アプローチ（2/2）}
平均分散モデルとは以下のようなモデルである．
\begin{eqnarray*}
		%\mbox{minimize} && \sum^{n}_{i=1} \sum^{n}_{j=1} {\sigma_{ij}w_i w_j} \\
             \mbox{minimize} && \sigma_p^2, \\
		\mbox{subject to}  &&R_p \geq R_e, \\
						&&\sum^{n}_{i=1} {w_i} = 1,\\
						&&0 \leq w_{i} \leq 1.(i=1,\dots,n)
\end{eqnarray*}

\begin{itemize}
\item $\sigma_p^2$：ポートフォリオ全体の収益率の分散
%\item $\sigma_{ij}$：資産$i$と資産$j$の収益率の共分散
%\item $w_i$：資産$i$のポートフォリオウェイト
%\item $w_j$：資産$j$のポートフォリオウェイト
\item $R_p$：ポートフォリオの収益率
\item $R_e$：要求期待収益率
\item $w_i$：資産$i$のポートフォリオウェイト
\item $n$：資産数
\end{itemize}
\end{frame}
